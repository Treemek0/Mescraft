#version 330 core
layout(location = 0) in vec2 vertexPos;
layout(location = 1) in vec2 vertexUV;

out vec2 TexCoord;

uniform mat4 projection;
uniform vec2 offset;   // position offset
uniform vec2 scale;    // size scaling

void main()
{
    vec2 pos = vertexPos * scale + offset;
    gl_Position = projection * vec4(pos, 0.0, 1.0);
    TexCoord = vertexUV;
}
