#version 330 core
out vec4 FragColor;

in vec2 TexCoord;
flat in float light;

uniform sampler2D textureSampler;

void main()
{
    vec4 texColor = texture(textureSampler, TexCoord);

    // Normalize the light value from [0, 255] to [0.0, 1.0]
    FragColor = texColor * (light / 255.0);
}
