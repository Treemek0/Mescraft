#version 330 core
out vec4 FragColor;

in vec2 TexCoord;
flat in float FaceType;

uniform sampler2D textureSampler;

void main()
{
    vec4 texColor = texture(textureSampler, TexCoord);
    
    float brightness = 1.0;

    if(FaceType < 4.0){ // side
        brightness *= 0.8;
    }else if(FaceType == 5){ // bottom
        brightness *= 0.65;
    }

    FragColor = texColor * brightness;
}
